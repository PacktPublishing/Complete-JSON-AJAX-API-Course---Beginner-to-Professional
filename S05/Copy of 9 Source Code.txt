        function webData() {
            var url = 'https://api.spacexdata.com/v2/launches/'
            $.get(url,output)
        }
        function output(data) {
            console.log(data[1])
            $('#output').html('<h2>' + data[1].flight_number + '</h2><p>' + data[1].details + '</p>')
        }
